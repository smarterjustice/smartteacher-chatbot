    <button onclick="sendMessage()">Send</button>

<script>
    async function sendMessage() {
        const userInput = document.getElementById('userInput');
        const chatLog = document.getElementById('chatLog');

        if (!userInput.value.trim()) return;

        chatLog.innerHTML += `<div class="user-message">${userInput.value}</div>`;

        const response = await fetch('https://api.openai.com/v1/chat/completions', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': 'Bearer YOUR_OPENAI_API_KEY'
            },
            body: JSON.stringify({
                model: 'gpt-4-turbo',
                messages: [
                    {"role": "system", "content": "You are SmartTeacher, a Legal English classroom assistant. Provide precise, clear translations, legal grammar help, contract explanations, and plain-English rewriting. Answer formally, clearly, and never offer legal advice."},
                    {"role": "user", "content": userInput.value}
                ]
            })
        });

        const data = await response.json();
        const botMessage = data.choices[0].message.content;

        chatLog.innerHTML += `<div class="bot-message">${botMessage}</div>`;
        chatLog.scrollTop = chatLog.scrollHeight;
        userInput.value = '';
    }
</script>
